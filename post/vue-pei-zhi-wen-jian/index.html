	'vue': 'Vue',
	'vue-router': 'VueRouter',
	'vuex': 'Vuex',
	// 'axios': 'axios',
	// 'lodash': '_',

	// 'es6-promise': 'es6-promise',
	// 'vue-lazyload': 'VueLazyload',

postcss.config.js

配置px 转rem 和自动补全前缀

module.exports = {
  "plugins": {
    "autoprefixer": {},
    "postcss-import": {},
    'autoprefixer': {
      browsers: ['Android >= 4.0', 'iOS >= 7']
    },
    'postcss-pxtorem': {
      rootValue: 19,
      propList: ['*']
    }
  }
}
